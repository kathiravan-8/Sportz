// ======================================================
// RENDER
// ======================================================
function renderSport(sport) {
  const data = sportsData[sport];
  if (!data) return;

  const container = document.getElementById('sportsContent');
  container.innerHTML = '';

  data.sections.forEach((section, i) => {
    const block = document.createElement('div');
    block.className = 'section-block fade-in';
    block.style.animationDelay = `${i * 0.1}s`;

    block.innerHTML = `
      <div class="section-title">${section.title}</div>
      <p class="scroll-hint">← scroll to see more →</p>
      <div class="product-row">
        ${section.products.map(p => createProductCard(p)).join('')}
      </div>
    `;
    container.appendChild(block);
  });
}

function createProductCard(product) {
  const inCart = cart.some(c => c.id === product.id);
  return `
    <div class="product-card" onclick="openModal('${product.id}')">
      <div class="product-img-wrap">
        <img src="${product.img}" alt="${product.name}" loading="lazy" onerror="this.src='https://via.placeholder.com/160x160/1a1a1a/888?text=IMG'">
        <button class="add-btn ${inCart ? 'added' : ''}" id="addbtn-${product.id}"
          onclick="event.stopPropagation(); addToCart('${product.id}')">${inCart ? 'ADDED' : 'Add+'}</button>
      </div>
      <div class="product-name">${product.name}</div>
      <div class="product-sub">${product.sub}</div>
      <div class="product-price">₹${product.price.toLocaleString()}</div>
      <div class="view-details" onclick="event.stopPropagation(); openModal('${product.id}')">View Details</div>
    </div>
  `;
}

function selectSport(sport, tabEl) {
  currentSport = sport;

  // Update tabs
  document.querySelectorAll('.sport-tab').forEach(t => t.classList.remove('active'));
  if (tabEl) tabEl.classList.add('active');
  else {
    document.querySelectorAll('.sport-tab').forEach(t => {
      if (t.textContent.toLowerCase() === sport) t.classList.add('active');
    });
  }

  // Update hero tiles
  document.querySelectorAll('.sport-tile').forEach(t => t.classList.remove('active'));
  const tile = document.getElementById('tile-' + sport);
  if (tile) tile.classList.add('active');

  renderSport(sport);
}

function setActiveTile(el) {
  document.querySelectorAll('.sport-tile').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
}

}

function findProduct(id) {
  for (const sport of Object.values(sportsData)) {
    for (const section of sport.sections) {
      const p = section.products.find(p => p.id === id);
      if (p) return p;
    }
  }
  return null;
}

function addToCart(id) {
  const product = findProduct(id);
  if (!product) return;

  const existing = cart.find(c => c.id === id);
  if (existing) {
    existing.qty++;
  } else {
    cart.push({ ...product, qty: 1 });
  }

  updateCartBadge();
  updateAddBtn(id, true);
  showToast(`${product.name} added to cart!`);
  renderCartItems();
}

function addToCartFromModal() {
  if (modalProduct) {
    addToCart(modalProduct.id);
    closeModal();
  }
}

function removeFromCart(id) {
  cart = cart.filter(c => c.id !== id);
  updateCartBadge();
  updateAddBtn(id, false);
  renderCartItems();
}

function updateQty(id, delta) {
  const item = cart.find(c => c.id === id);
  if (!item) return;
  item.qty += delta;
  if (item.qty <= 0) removeFromCart(id);
  else renderCartItems();
}

function updateAddBtn(id, added) {
  const btn = document.getElementById('addbtn-' + id);
  if (btn) {
    btn.textContent = added ? 'ADDED' : 'Add+';
    btn.classList.toggle('added', added);
  }
}

function updateCartBadge() {
  const total = cart.reduce((sum, i) => sum + i.qty, 0);
  document.getElementById('cartBadge').textContent = total;
}

function renderCartItems() {
  const container = document.getElementById('cartItems');
  const footer = document.getElementById('cartFooter');

  if (cart.length === 0) {
    container.innerHTML = `
      <div class="empty-cart">
        <svg width="64" height="64" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1">
          <path d="M6 2 3 6v14a2 2 0 0 0 2 2h14a2 2 0 0 0 2-2V6l-3-4z"/><line x1="3" y1="6" x2="21" y2="6"/><path d="M16 10a4 4 0 0 1-8 0"/>
        </svg>
        <p>YOUR CART IS EMPTY</p>
      </div>`;
    footer.style.display = 'none';
    return;
  }

  const total = cart.reduce((sum, i) => sum + i.price * i.qty, 0);

  container.innerHTML = cart.map(item => `
    <div class="cart-item">
      <div class="cart-item-img">
        <img src="${item.img}" alt="${item.name}" onerror="this.src='https://via.placeholder.com/56x56/222/888?text=IMG'">
      </div>
      <div class="cart-item-info">
        <div class="cart-item-name">${item.name}</div>
        <div class="cart-item-price">₹${(item.price * item.qty).toLocaleString()}</div>
        <div class="qty-control">
          <button class="qty-btn" onclick="updateQty('${item.id}', -1)">−</button>
          <span class="qty-num">${item.qty}</span>
          <button class="qty-btn" onclick="updateQty('${item.id}', 1)">+</button>
        </div>
      </div>
      <button class="cart-item-remove" onclick="removeFromCart('${item.id}')">✕</button>
    </div>
  `).join('');

  document.getElementById('cartTotal').textContent = '₹' + total.toLocaleString();
  footer.style.display = 'block';
}

function openCart() {
  document.getElementById('cartOverlay').classList.add('open');
  document.getElementById('cartPanel').classList.add('open');
  renderCartItems();
}

function closeCart() {
  document.getElementById('cartOverlay').classList.remove('open');
  document.getElementById('cartPanel').classList.remove('open');
}

// MODAL
// ======================================================
function openModal(id) {
  const product = findProduct(id);
  if (!product) return;
  modalProduct = product;

  document.getElementById('modalImg').src = product.img;
  document.getElementById('modalName').textContent = product.name;
  document.getElementById('modalSub').textContent = product.sub;
  document.getElementById('modalPrice').textContent = '₹' + product.price.toLocaleString();

  const inCart = cart.some(c => c.id === id);
  document.getElementById('modalAddBtn').textContent = inCart ? 'ADDED TO CART ✓' : 'ADD TO CART';

  document.getElementById('modalOverlay').classList.add('open');
}

function closeModal(e) {
  if (!e || e.target === document.getElementById('modalOverlay')) {
    document.getElementById('modalOverlay').classList.remove('open');
    modalProduct = null;
  }
}

let searchTimeout;

function handleSearch(query) {
  clearTimeout(searchTimeout);
  searchTimeout = setTimeout(() => {
    if (!query.trim()) {
      hideDropdown();
      return;
    }
    const results = [];
    const q = query.toLowerCase();
    for (const sport of Object.values(sportsData)) {
      for (const section of sport.sections) {
        for (const p of section.products) {
          if (p.name.toLowerCase().includes(q) || p.sub.toLowerCase().includes(q)) {
            results.push(p);
          }
        }
      }
    }
    renderDropdown(results.slice(0, 6));
  }, 200);
}

function renderDropdown(results) {
  const dd = document.getElementById('searchDropdown');
  if (!results.length) {
    dd.innerHTML = '<div style="padding:12px;font-size:0.75rem;color:var(--muted);font-family:Space Mono,monospace;">No results found</div>';
  } else {
    dd.innerHTML = results.map(p => `
      <div class="search-result-item" onmousedown="openModal('${p.id}')">
        <div class="search-result-img">
          <img src="${p.img}" alt="${p.name}" onerror="this.src='https://via.placeholder.com/40x40/222/888?text=IMG'">
        </div>
        <div>
          <div class="search-result-name">${p.name}</div>
          <div class="search-result-price">₹${p.price.toLocaleString()}</div>
        </div>
      </div>
    `).join('');
  }
  dd.classList.add('show');
}

function showDropdown() {
  const q = document.getElementById('searchInput').value;
  if (q) handleSearch(q);
}

function hideDropdown() {
  setTimeout(() => {
    document.getElementById('searchDropdown').classList.remove('show');
  }, 200);
}

let toastTimer;
function showToast(msg) {
  const toast = document.getElementById('toast');
  toast.textContent = msg;
  toast.classList.add('show');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => toast.classList.remove('show'), 2500);
}
